# CA
openssl req -new -x509 -days 365 -keyout ca.key -out ca.crt -subj "/CN=CA"

# Servidor
openssl req -new -keyout server.key -out server.csr -subj "/CN=postgres"
openssl x509 -req -in server.csr -CA ca.crt -CAkey ca.key -out server.crt -days 365

# Cliente (el backend)
openssl req -new -keyout client.key -out client.csr -subj "/CN=postgres"
openssl x509 -req -in client.csr -CA ca.crt -CAkey ca.key -out client.crt -days 365
